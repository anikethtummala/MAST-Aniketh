$$
\begin{aligned}
& A P+P X+X S+S C+C R+R Y+Y Q+Q B \\
= & A P+P S+S R+R Q+Q B \\
= & A P+t+t+t+Q B \\

= & c-t+3 t \\
= & c+2 t .
\end{aligned}
$$
